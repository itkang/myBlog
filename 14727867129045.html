使用github作为中转自动更新服务器中的文件

#! /bin/bash
cd /home/itkang/Blog;git pull origin master;git add .;git commit -m "upgrade";git push origin master; #本地文件提交到github
ssh $USER@$SERVER 'cd /home/myBlog;git pull origin master;' #远程服务器pull文件
echo 'all upgrade';
